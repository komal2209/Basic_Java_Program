class Solution {
    public int countPrimes(int n) {
        if(n<=1)
        return 0;
        if(n==2 || n==3)
         return 1;
     int count =0;
    boolean isPrime[]=new boolean[n+1];
    Arrays.fill(isPrime,true);
     for(int i=2;i*i<=n;i++)
     {
      if(isPrime[i]==true)
      {
         for(int j=i*i;j<=n;j=j+i)
         {
         isPrime[j]=false;
         }
      }
     }
for(int k=2; k<=n;k++)
{
if(isPrime[k])
count++;
}
return count;
    }
}
